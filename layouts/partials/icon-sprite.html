{{- /* Builds a single external SVG sprite from every assets/images/*.svg file and returns its (fingerprinted) URL. Each icon becomes a . Referenced via in partials/svg.html, so the icon markup leaves the page HTML and the sprite is fetched once and cached. Call through partialCached so it is built only once per site build. */ -}} {{- $symbols := slice -}} {{- range (resources.Match "images/*.svg") -}} {{- $name := strings.TrimSuffix ".svg" (path.Base .Name) -}} {{- $c := .Content -}} {{- /* drop XML declaration and leading comments */ -}} {{- $c = replaceRE `(?s)^\s*(?:<\?xml[^>]*>)?\s*(?:\s*)*` "" $c -}} {{- $vb := "" -}} {{- with (findRE `viewBox="[^"]*"` $c 1) -}}{{- $vb = index . 0 -}}{{- end -}} {{- $inner := replaceRE `(?s)^.*?]*>(.*)\s*$` "$1" $c -}} {{- $symbols = $symbols | append (printf `%s` $name $vb $inner) -}} {{- end -}} {{- $svg := printf `` (delimit $symbols "") -}} {{- $res := resources.FromString "icons.svg" $svg | minify | fingerprint -}} {{- return $res.RelPermalink -}}
